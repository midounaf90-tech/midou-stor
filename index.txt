<script>
  /* =========================================================
     CONFIG — عدّل هذا الرابط فقط بعد إنشاء حساب SheetDB
     ========================================================= */
  const SHEETDB_API = "https://sheetdb.io/api/v1/9jjeye7lrsnri";

  let currentProduct = null;

  // ====== Load products from Google Sheet (tab "Products") ======
  async function loadProducts(){
    const grid = document.getElementById('grid');
    try{
      const res = await fetch(SHEETDB_API);
      const data = await res.json();

      if(!Array.isArray(data) || data.length === 0){
        grid.innerHTML = `<div class="empty-state">لا توجد منتجات حالياً</div>`;
        document.getElementById('prod-count').textContent = "0 منتج";
        return;
      }

      document.getElementById('prod-count').textContent = `${data.length} منتج`;

      grid.innerHTML = data.map(p => `
        <div class="card">
          <div class="card-img">
            ${p.image ? `<img src="${p.image}" alt="${p.name}" onerror="this.parentElement.innerHTML='<div class=\\'ph\\'>📦</div>'">` : `<div class="ph">📦</div>`}
          </div>
          <div class="card-body">
            <h3>${p.name}</h3>
            <p>${p.description || ''}</p>
            <div class="price-row">
              <span class="price">${p.price}</span>
              <button class="order-btn" onclick='openModal(${JSON.stringify(p.name)}, ${JSON.stringify(p.price)})'>اطلب الآن</button>
            </div>
          </div>
        </div>
      `).join('');

    }catch(err){
      grid.innerHTML = `<div class="empty-state">تعذر تحميل المنتجات. تحقق من رابط SheetDB.</div>`;
      console.error(err);
    }
  }

  function openModal(name, price){
    currentProduct = {name, price};
    document.getElementById('modal-product-name').textContent = name;
    document.getElementById('modal-product-price').textContent = `السعر: ${price} دج`;
    document.getElementById('f-name').value = '';
    document.getElementById('f-phone').value = '';
    document.getElementById('f-address').value = '';
    document.getElementById('status-msg').textContent = '';
    document.getElementById('status-msg').className = 'status-msg';
    document.getElementById('overlay').classList.add('active');
  }

  function closeModal(){
    document.getElementById('overlay').classList.remove('active');
  }

  // ====== Submit order to Google Sheet (tab "Orders") ======
  async function submitOrder(){
    const name = document.getElementById('f-name').value.trim();
    const phone = document.getElementById('f-phone').value.trim();
    const address = document.getElementById('f-address').value.trim();
    const statusEl = document.getElementById('status-msg');
    const btn = document.getElementById('submit-btn');

    if(!name || !phone || !address){
      statusEl.textContent = "عمر جميع الخانات من فضلك";
      statusEl.className = "status-msg err";
      return;
    }

    btn.disabled = true;
    btn.textContent = "جاري الإرسال...";

    const order = {
      name: name,
      phone: phone,
      address: address,
      product: currentProduct.name,
      price: currentProduct.price,
      date: new Date().toLocaleString('fr-FR', {timeZone:'Africa/Algiers'})
    };

    try{
      const res = await fetch(`${SHEETDB_API}?sheet=Orders`, {
        method: 'POST',
        headers: {'Content-Type':'application/json'},
        body: JSON.stringify({ data: [order] })
      });

      if(res.ok){
        statusEl.textContent = "✓ تم تسجيل طلبك بنجاح! نتصلو بيك قريباً.";
        statusEl.className = "status-msg ok";
        setTimeout(closeModal, 1800);
      }else{
        throw new Error('Failed');
      }
    }catch(err){
      statusEl.textContent = "✕ خطأ، حاول مرة أخرى";
      statusEl.className = "status-msg err";
    }finally{
      btn.disabled = false;
      btn.textContent = "تأكيد الطلب";
    }
  }

  // Close modal on outside click
  document.getElementById('overlay').addEventListener('click', (e) => {
    if(e.target.id === 'overlay') closeModal();
  });

  loadProducts();
</script>
